  Ecco i valori raggruppati: 53 59 41 31 87 37 22 26 80 78

#define dim 10

void leggi(int a[]) {
  for (int i = 0; i < dim; i++)
    a[i] = rand() % 90 + 1;
}

void partiziona(int a[], int n) {
  int temp = 0;
  int j = 0;
  
  for (int i = 0; i < n; i++) {
      if (a[i] % 2 != 0) {
        temp = a[i];
        a[i] = a[j];
        a[j] = temp;
        j ++;
    }
  }

  for (int i = 0; i < dim; i++)
    printf("%d ", a[i]);
}

int main(void) {
  srand(time(NULL));
  int a[dim];

  leggi(a);

  for (int i = 0; i < dim; i++)
    printf("%d ", a[i]);

  printf("\n");

  partiziona(a, dim);

  // printf("Hello World\n");
  return 0;
}
